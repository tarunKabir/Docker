#Docker file

FROM ubuntu:15:04       // FROM creates the layer from ubuntu 
COPY . /app             // copy adds files from your Docker client's current directory 
RUN make /app           // builds your application with make 
CMD python /app/app.py  // cmd specifies what command to run within container


## 
FROM : Initialises  a new build image and sets the base image
RUN  :  Will execute any commands in a new layer
CMD  :  Provides a default for an executing container.
       There can only be one CMD instruction in a docker file.
LABEL : Adds metadata to an image
EXPOSE :  Informs the docker that the container listens to the specified network ports at runtime
ENV : sets the environment variable <key> to the <value>
ADD : Copies new files , directories or remote URLs from <src> and adds them to  the filesystem of the image
at the path <dest>
